023     */
024    public void ramp(double want)
025    {
026        double cur = get(); // cur = current speed
027        
028        if(Math.abs(want - cur) > Config.rampRate)
029            want = cur + (Config.rampRate * want > cur ? 1: -1);
030        
031        set(want);
032    }
