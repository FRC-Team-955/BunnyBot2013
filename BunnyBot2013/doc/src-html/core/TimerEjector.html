005package core;
006
007import util.MyTalon;
008import edu.wpi.first.wpilibj.Timer;
009import util.Config;
010import util.MyJoystick;
011import util.Output;
012
013/**
014 * @author Trevor Controls the ejector with a timer
015 */
016public class TimerEjector {
017
018        protected MyJoystick joy;
019        protected int time = Config.ejectorTime;// in milliseconds
020        protected double speed = Config.ejectorSpeed;
021        protected MyTalon ejectorTalon = new MyTalon(Config.chnEjector);
022        protected Timer timer = new Timer();
023        protected boolean isRetracting;
024        protected boolean isRetracted = true;
025        protected boolean isExtending;
026
027        public TimerEjector(MyJoystick joy1) {
028                joy = joy1;
029        }
030
031        /**
032         * Starts the ejector and the timer. It has 3 states, isRetracted,
033         * isExtending, and isRetracting
034         */
035        public void run() {
036                //Managing state transitions between 3 states. The three states are Retracted, Extending, and Retracting
037                //If the ejector isRetracted and the button is pushed start it and se isExtending
038                if (isRetracted && joy.getButton(Config.btEjector)) {
039                        timer.start();
040                        ejectorTalon.set(speed);
041                        isRetracting = false;
042                        isRetracted = false;
043                        isExtending = true;
044
045                        Output.println(Config.IdEjector, "Ejector Active!" + true);
046                } //If the ejector goes for the set amount of time, have it go back and set isRetracting true
047                else if (isExtending) {
048                        if (timer.get() >= time) {
049                                ejectorTalon.set(-speed);
050                                isRetracting = true;
051                                isRetracted = false;
052                                isExtending = false;
053                                Output.println(Config.IdEjector, "Ejector Retracting!");
054                        }
055                } //if the ejector isRetracting, and it has retracted for the set amount of time, then stop the ejector,
056                //reset the timer, and set isRetracted.
057                else if (isRetracting) {
058                        if (timer.get() >= time * 2) {
059                                ejectorTalon.set(0);
060                                timer.stop();
061                                timer.reset();
062                                isRetracted = true;
063                                isRetracting = false;
064                                isExtending = false;
065                                Output.println(Config.IdEjector, "Ejector Retracted!");
066                        }
067                } else {
068                        ejectorTalon.set(0);
069                        timer.stop();
070                        timer.reset();
071                        isRetracted = true;
072                        isRetracting = false;
073                        isExtending = false;
074                        Output.println(Config.IdEjector, "Invalid state in TimerEjector");
075                }
076        }
077}
