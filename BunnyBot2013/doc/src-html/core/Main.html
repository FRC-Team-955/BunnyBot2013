007package core;
008
009import util.Output;
010import edu.wpi.first.wpilibj.IterativeRobot;
011import util.Config;
012import util.MyJoystick;
013import auto.Autonomous;
014/**
015 * The VM is configured to automatically run this class, and to call the
016 * functions corresponding to each mode, as described in the IterativeRobot
017 * documentation. If you change the name of this class or the package after
018 * creating this project, you must also update the manifest file in the resource
019 * directory.
020 */
021public class Main extends IterativeRobot {
022
023    MyJoystick joy;
024    LimitEjector eject;
025    Drive drive;
026    Autonomous auto;
027    
028    /**
029     * This function is run when the robot is first started up and should be
030     * used for any initialization code.
031     */
032    public void robotInit() 
033        {
034                System.out.println("Bunny Bot 2013");
035        Output.updateArray();
036        joy = new MyJoystick(Config.chnJoystick, Config.buttonsOnJoystick);
037        joy.setAxisChannel(MyJoystick.AxisType.kX, 3);
038        joy.setAxisChannel(MyJoystick.AxisType.kY, 2);
039        eject = new LimitEjector(joy);
040        drive = new Drive(joy);
041        auto = new Autonomous(joy);
042        
043        }
044
045    /**
046     * This function is called periodically during autonomous
047     */
048    public void autonomousPeriodic() 
049        {
050        Output.updateArray();
051        auto.replay();
052        eject.run();
053        drive.run();
054        
055        }
056
057    /**
058     * This function is called periodically during operator control
059     */
060    public void teleopPeriodic() {
061        Output.updateArray();
062                joy.updateButtons();
063        eject.run();
064        drive.run();
065        auto.run();
066        }
067
068    /**
069     * Called once in m_autonomous
070     */
071    public void autonomousInit() {
072        Output.updateArray();
073        auto.setFile();
074        
075        }
076
077    /**
078     * This function is called once when the robot is disabled
079     */
080    public void disabledInit() {
081        Output.updateArray();
082
083        // Resets the replay to false if it was true before
084        auto.resetAutonomous();
085    }
086}
