005package util;
006
007import edu.wpi.first.wpilibj.Talon;
008
009/**
010 * @author seraj.bonakdar
011 * Extends talon with ramp function
012 */
013public class MyTalon extends Talon {
014
015    public MyTalon(int channel) {
016        super(channel);                  
017        }
018    
019    /**
020     * Checks if the talon is attempting to be set to a positive or a negative
021     * number the makes sure it within the allowed distance. 
022     * @param want the speed that the joystick is trying to set the talon to.
023     */
024    public void ramp(double want)
025    {
026        double cur = get(); // cur = current speed
027        
028        if(Math.abs(want - cur) > Config.rampRate)
029            want = cur + (Config.rampRate * (want > cur ? 1: -1));
030        
031        set(want);
032    }
033}
