006package util;
007
008/**
009 * Holds data corresponding to the values of the robot.
010 *
011 * @author Fauzi
012 */
013public class JoyData {
014
015    private double m_time = 0;
016    private double m_x = 0;
017    private double m_y = 0;
018    private boolean m_push = false;
019
020    /**
021     * Sets all the properties
022     * @param timer
023     * @param x
024     * @param y
025     * @param ejector 
026     */
027    private void setValues(double timer, double x, double y, boolean ejector) {
028        m_time = timer;
029        m_x = x;
030        m_y = y;
031        m_push = ejector;
032    }
033
034    /**
035     * Sets all the properties
036     * @param time
037     * @param joy 
038     */
039    public void setValues(double time, MyJoystick joy) {
040        setValues(time, joy.getMyX(), joy.getMyY(), joy.getButton(Config.btEjector));
041    }
042
043    /**
044     * Sets all the properties
045     * @param emu 
046     */
047    public void setValues(JoyData emu) {
048        setValues(emu.getTimer(), emu.getX(), emu.getY(), emu.getEjector());
049    }
050
051    /** 
052     * Sets the time stamp of when the data was taken
053     * @param timeVal 
054     */
055    public void setTime(double timeVal) {
056        m_time = timeVal;
057    }
058
059    /**
060     * Sets the x property of the joystick data
061     * @param x
062     */
063    public void setX(double x) {
064        m_x = x;
065    }
066
067    /**
068     * Sets the y property of the joystick data
069     * @param y 
070     */
071    public void setY(double y) {
072        m_y = y;
073    }
074
075    /**
076     * Sets the ejector
077     * @param val 
078     */
079    public void setEjector(boolean val) {
080        m_push = val;
081    }
082
083    /**
084     * Returns time stamp of the data.
085     * @return
086     */
087    public double getTimer() {
088        return m_time;
089    }
090
091    /**
092     * Gets the x value.
093     * @return
094     */
095    public double getX() {
096        return m_x;
097    }
098
099    /**
100     * Gets the y value.
101     * @return
102     */
103    public double getY() {
104        return m_y;
105    }
106
107    /**
108     * Gets the ejector status
109     * @return 
110     */
111    public boolean getEjector() {
112        return m_push;
113    }
114}
