007package util;
008
009import edu.wpi.first.wpilibj.DriverStation;
010import edu.wpi.first.wpilibj.DriverStationLCD;
011
012/**
013 *
014 * @author Merfoo
015 */
016public class Station 
017{
018    /**
019     * Gets the button status from the driverstation, 1 - 8 available.
020     * @param iChan
021     * @return 
022     */
023    public static boolean getDitigalIn(int iChan)
024    {
025        return DriverStation.getInstance().getDigitalIn(iChan);
026    }
027    
028        /**
029     * Gets the analog value from the driverstation, 1 - 8 available.
030     * @param iChan
031     * @return 
032     */
033    public static double getAnalogIn(int iChan)
034    {
035        return DriverStation.getInstance().getAnalogIn(iChan);
036    }
037        
038    /**
039     * Prints specified message to the driver station on the corresponding line
040     * 1-6 are available.
041     */
042    public static void print(int iLine, String sMessage)
043    {
044        DriverStationLCD.Line line = null;
045        
046        switch(iLine)
047        {
048            case 1:
049            {
050                line = DriverStationLCD.Line.kUser1;
051                break;
052            }
053            
054            case 2:
055            {
056                line = DriverStationLCD.Line.kUser2;
057                break;
058            }
059                
060            case 3:
061            {
062                line = DriverStationLCD.Line.kUser3;
063                break;
064            }
065                
066            case 4:
067            {
068                line = DriverStationLCD.Line.kUser4;
069                break;
070            }
071                
072            case 5:
073            {
074                line = DriverStationLCD.Line.kUser5;
075                break;
076            }
077                
078            case 6:
079            {
080                line = DriverStationLCD.Line.kUser6;
081                break;
082            }
083        }
084        
085        DriverStationLCD.getInstance().println(line, 1, sMessage);
086        DriverStationLCD.getInstance().updateLCD(); 
087    }
088}
