006package util;
007
008/**
009 * @author Pedro Custom math functions: round and power.
010 */
011public class MyMath {
012
013    /**
014     * returns num rounded to x digits.
015     *
016     * @param num
017     * @param digits
018     * @return
019     */
020    public static double round(double num, int digits) {
021        double scalar = power(10, digits);
022        num *= scalar;
023        double dec = num - Math.floor(num);
024        num = Math.floor(num);
025
026        if (dec >= .5) {
027            num++;
028        }
029
030        num /= scalar;
031        return num;
032    }
033
034    /**
035     * Returns a to the power of b.
036     *
037     * @param a
038     * @param b
039     * @return
040     */
041    public static double power(double a, double b) {
042        double temp = a;
043
044        for (int i = 1; i < b; i++) {
045            a *= temp;
046        }
047
048        if (b == 0) {
049            return 1;
050        }
051
052        return a;
053    }
054
055    /**
056     * Rounds the double to the hundreth's place, ex. 12.345 -> 12.35.
057     * @param value Value to be rounded
058     * @return 
059     */
060    public static double round(double value) {
061        return Math.floor(value * 100 + 0.5) / 100;
062    }
063}
